# 2. Import vLLM
from vllm import LLM, SamplingParams
import time

# 3. Define prompts
prompts = [
    "The sun seems warmer today",
    "I hope we can go out ",
    "Are you listening?",
]

# 4. Define sampling parameters
sampling_params = SamplingParams(
    temperature=0.8,
    top_p=0.95,
    max_tokens=50,  # number of tokens to generate
)

# 5. Run generation
llm1 = LLM(model="TinyLlama/TinyLlama-1.1B-Chat-v1.0")

print("Generating completions with TinyLlama (vLLM)...")
start_time_tinyllama = time.time()
outputs_tinyllama = llm1.generate(prompts, sampling_params)
end_time_tinyllama = time.time()
tinyllama_inference_time = end_time_tinyllama - start_time_tinyllama

# 6. Store and Print results
tinyllama_completions = []
for i, output in enumerate(outputs_tinyllama):
    print(f"\nPrompt {i+1}: {prompts[i]}")
    print(f"Completion (TinyLlama): {output.outputs[0].text}")
    tinyllama_completions.append(output.outputs[0].text)

print(f"\nTinyLlama Inference Time: {tinyllama_inference_time:.2f} seconds")
